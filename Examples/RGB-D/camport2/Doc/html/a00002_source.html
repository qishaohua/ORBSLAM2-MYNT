8 // README:
9 //------------------------------------------------------------------------------
10 //
11 // Depth camera, called "device", consists of several components. Each component
12 // is a hardware module or virtual module, such as RGB sensor, depth sensor.
13 // Each component has its own features, such as image resolution, pixel format.
14 //
15 // NOTE: The component TY_COMPONENT_DEVICE is a virtual component that contains
16 // all features related to the whole device, such as trigger mode, device IP.
17 //
18 // Each frame consists of several images. Normally, all the images have identical
19 // timestamp, means they are captured at the same time.
20 //
21 //------------------------------------------------------------------------------
